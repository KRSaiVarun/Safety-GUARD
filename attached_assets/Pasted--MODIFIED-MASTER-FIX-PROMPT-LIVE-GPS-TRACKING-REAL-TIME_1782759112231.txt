// In emergency/useGPS.ts or similar
const startGPSTracking = () => {
  if (!navigator.geolocation) {
    console.error('[GPS] Not supported');
    return;
  }

  navigator.geolocation.watchPosition(
    (position) => {
      const { latitude, longitude, accuracy, speed, heading } = position.coords;
      console.log('[GPS] Update:', { latitude, longitude, accuracy });
      
      // Store in Zustand
      useEmergencyStore.getState().updateLocation({
        latitude,
        longitude,
        accuracy,
        speed,
        heading,
        timestamp: new Date().toISOString()
      });
      
      // Emit via Socket
      socket.emit('LOCATION_UPDATED', {
        sessionId: emergencySession.id,
        userId: user.id,
        latitude,
        longitude,
        accuracy,
        speed,
        heading,
        battery: navigator.getBattery ? await navigator.getBattery().then(b => b.level) : null,
        timestamp: new Date().toISOString(),
        status: 'ACTIVE'
      });
    },
    (error) => {
      console.error('[GPS] Error:', error.message);
      // Show permission UI if denied
      if (error.code === 1) {
        setShowPermissionRequest(true);
      }
    },
    {
      enableHighAccuracy: true,
      maximumAge: 0,
      timeout: 5000
    }
  );
};

# In socket_manager.py or similar
@sio.on('LOCATION_UPDATED')
async def handle_location_update(sid, data):
    print(f'[BACKEND] Received: {data}')
    
    # Update database
    async with db_pool.acquire() as conn:
        await conn.execute(
            """
            UPDATE emergency_sessions 
            SET 
                latitude = $1,
                longitude = $2,
                accuracy = $3,
                speed = $4,
                heading = $5,
                battery = $6,
                last_updated = NOW(),
                gps_points = gps_points || $7::jsonb
            WHERE session_id = $8
            """,
            data['latitude'],
            data['longitude'],
            data['accuracy'],
            data['speed'],
            data['heading'],
            data['battery'],
            json.dumps({'lat': data['latitude'], 'lng': data['longitude'], 'timestamp': data['timestamp']}),
            data['sessionId']
        )
    
    # Broadcast to all admin dashboards
    await sio.emit('LOCATION_UPDATED', data, room='dashboard')
    print(f'[BACKEND] Broadcast to dashboard room')

D. Map Component (React/TypeScript)
```typescript
// In EmergencyMap.tsx
const EmergencyMap = () => {
  const location = useDashboardStore(state => state.currentLocation);
  const mapRef = useRef<LeafletMap>(null);
  const markerRef = useRef<L.Marker>(null);
  const polylineRef = useRef<L.Polyline>(null);
  const pointsRef = useRef<[number, number][]>([]);
  
  // Use live location or fallback
  const center = location 
    ? [location.latitude, location.longitude] 
    : [27.8974, 78.0880]; // Aligarh fallback
  
  console.log('[MAP] Center:', center, 'Live:', !!location);
  
  // Fly to location when live data arrives
  useEffect(() => {
    if (location && mapRef.current) {
      console.log('[MAP] Flying to:', location);
      mapRef.current.flyTo([location.latitude, location.longitude], 17);
      
      // Update marker
      if (markerRef.current) {
        markerRef.current.setLatLng([location.latitude, location.longitude]);
      } else {
        // Create marker with SOS badge
        const icon = L.divIcon({
          html: `
            <div class="sos-marker">
              <img src="${user.profilePhoto}" class="profile-image" />
              <div class="sos-badge">SOS</div>
              <div class="pulse-ring"></div>
            </div>
          `,
          className: 'custom-marker',
          iconSize: [50, 50]
        });
        
        markerRef.current = L.marker([location.latitude, location.longitude], { icon })
          .addTo(mapRef.current);
      }
      
      // Update polyline
      pointsRef.current.push([location.latitude, location.longitude]);
      if (pointsRef.current.length > 100) {
        pointsRef.current.shift();
      }
      
      if (polylineRef.current) {
        polylineRef.current.setLatLngs(pointsRef.current);
      } else {
        polylineRef.current = L.polyline(pointsRef.current, {
          color: 'red',
          weight: 3,
          opacity: 0.8
        }).addTo(mapRef.current);
      }
    }
  }, [location]);
  
  return (
    <MapContainer center={center} zoom={13} ref={mapRef}>
      <TileLayer url="https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png" />
      {/* Rest of map */}
    </MapContainer>
  );
};
```
